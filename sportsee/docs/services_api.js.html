import axios from 'axios';

/**
 * Legacy code of initial API data fetching
 */
async function resolve(promise) {
    const resolved = {
      data: null,
      error: null
    };
  
    try {
      resolved.data = await promise;
    } catch(e) {
      resolved.error = e;
    }
  
    return resolved;
}

export async function validateUserExists(userId) {
  try{
    return await resolve(axios.get(`http://localhost:3000/user/${userId}`).then((res) => res.data === null ? false: true));
  } catch(err){
    return false;
  }
}

export async function getUser(userId) {
  return await resolve(axios.get(`http://localhost:3000/user/${userId}`).then(res => res.data));
}

export async function getFirstName(userId) {
  return await resolve(axios.get(`http://localhost:3000/user/${userId}`).then(res => res.data.data.userInfos.firstName));
}
